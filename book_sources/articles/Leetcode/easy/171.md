## 初見
 - 26進位轉10進位
```javascript=
var titleToNumber = function(columnTitle) {
    //codePointAt 'A' is 65
    columnTitle = columnTitle.split('');
    var pow = 0;
    var result = 0;
    while(columnTitle.length > 0){
        var chr = columnTitle.pop();
        result += Math.pow(26,pow)*(chr.codePointAt(0)-64);
        pow++;
    }
    return result;
};
```
