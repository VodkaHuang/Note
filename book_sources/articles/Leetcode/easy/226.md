## 初見
 - 使用遞迴實作反轉
```javascript=
/**
 * @param {TreeNode} root
 * @return {TreeNode}
 */
var invertTree = function(root) {
    if(!root) return null;
    let leftNode = root.left;
    let rightNode = root.right;
    root.left = invertTree(rightNode);
    root.right = invertTree(leftNode);
    return root;
};
```
