## 初見
 - solve by map+loop
```javascript=
var singleNumber = function(nums) {
    var map = new Map();
    var result;
    for(var i=0; i<nums.length; i++){
        map.set(nums[i],map.has(nums[i])?2:1);
    }
    map.forEach(function(val,key){
        if(val==1) result = key;
    });
    return result;
};
```
